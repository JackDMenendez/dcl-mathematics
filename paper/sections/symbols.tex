% ── Lattice and ambient spaces ─────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Lattice and ambient spaces}} \\
$d$
    & Lattice dimension; free parameter throughout the paper.  $d \geq 2$; specialised to $d = 3$ when matching Paper~I's physical setting.
    & \S\ref{sec:lattice_n_dim} \\
$\Lambda_d$
    & The diamond progression in dimension $d$: the bipartite octahedral lattice generated by the d-of-(d+1) regular simplex vertex construction.
    & \S\ref{sec:lattice_n_dim} \\
$V_d$
    & Vertex set of $\Lambda_d$, identified with $\mathbb{Z}^d$.
    & \S\ref{subsec:diamond_progression_construction} \\
$V_d^+, V_d^-$
    & The two colour classes of the bipartition; $V_d^+ = \{x : \sum_i x_i \text{ even}\}$, $V_d^-$ the odd-parity sites.
    & \S\ref{subsec:diamond_progression_construction} \\
$E_d$
    & Edge set of $\Lambda_d$ -- generated by the d-of-(d+1) simplex vertex offsets.
    & \S\ref{subsec:diamond_progression_construction} \\
$\epsilon$
    & Lattice-spacing parameter ($\epsilon > 0$); $\epsilon \to 0$ gives the discrete-to-continuum limit.
    & \S\ref{subsec:limit_parameter} \\
$\epsilon \Lambda_d$
    & Rescaled lattice $\subset \mathbb{R}^d$ with spacing $\epsilon$.
    & \S\ref{subsec:limit_parameter} \\
$\mathbb{R}^d$, $\mathbb{Z}^d$
    & Continuum ambient space; integer lattice (used as vertex-set model for $\Lambda_d$).
    & \S\ref{subsec:diamond_progression_construction} \\

% ── Three theorems ─────────────────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Three result-density theorems (\S\ref{sec:three_theorems})}} \\
$\text{coord}(d)$
    & Coordination number of $\Lambda_d$.  Theorem~\ref{thm:coord_2d}: $\text{coord}(d) = 2d$.
    & \S\ref{subsec:theorem_1} \\
$\sqrt{d}$
    & Maximum Euclidean hop in $\Lambda_d$ under the standard simplex projection.  Theorem~\ref{thm:max_hop}.
    & \S\ref{subsec:theorem_2} \\
$(d+1)^{(d-1)/2}$
    & Unit-cell volume of $\Lambda_d$.  Theorem~\ref{thm:unit_cell_volume}.
    & \S\ref{subsec:theorem_3} \\

% ── Symmetries ─────────────────────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Symmetries}} \\
$B_d$
    & Hyperoctahedral group: signed permutations of the $d$ coordinate axes; acts on $\Lambda_d$.
    & \S\ref{subsec:autom_generators} \\
$\mathfrak{a}_d$
    & The per-site automorphism algebra of $\Lambda_d$; provisional notation.  At $d = 3$, $\dim \mathfrak{a}_3 = 71$ (catalogued in \texttt{dcl-zoo}).
    & \S\ref{sec:automorphism_algebra} \\

% ── Discrete differential geometry ─────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Discrete differential geometry}} \\
$\Omega^k(\Lambda_d)$
    & Discrete $k$-forms on $\Lambda_d$: functions on oriented $k$-cells.
    & \S\ref{subsec:dec_forms} \\
$\mathrm{d}$
    & Discrete exterior derivative $\Omega^k \to \Omega^{k+1}$ (upright $\mathrm{d}$; avoids notational collision with the dimension $d$).  Satisfies $\mathrm{d} \circ \mathrm{d} = 0$.
    & \S\ref{subsec:dec_d} \\
$\mathrm{d}^\ast$
    & Discrete codifferential: formal adjoint of $\mathrm{d}$.
    & \S\ref{subsec:dec_codifferential} \\
$\langle \cdot, \cdot \rangle$
    & Inner product on $\Omega^k(\Lambda_d)$.
    & \S\ref{subsec:dec_codifferential} \\
$\Delta_{\Lambda_d} = -\mathrm{d}^\ast \mathrm{d} - \mathrm{d} \mathrm{d}^\ast$
    & Discrete Hodge Laplacian on $\Omega^k(\Lambda_d)$.
    & \S\ref{subsec:consistency_conditions} \\

% ── Function spaces ────────────────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Function spaces}} \\
$\ell^p(\Lambda_d)$
    & Discrete Lebesgue space, $1 \leq p \leq \infty$.
    & \S\ref{subsec:lp_spaces} \\
$\ell^{k,p}(\Lambda_d)$
    & Discrete Sobolev-analogue space: lattice functions with discrete derivatives (via $\mathrm{d}$) up to order $k$ in $\ell^p$.
    & \S\ref{subsec:sobolev_analogue} \\
$\|f\|_{p, \epsilon}$
    & $\epsilon$-rescaled $\ell^p$-norm: $\|f\|_{p, \epsilon}^p = \epsilon^d \sum_{x} |f(x)|^p$.
    & \S\ref{subsec:embeddings} \\
$L^p(\mathbb{R}^d)$, $W^{k,p}(\mathbb{R}^d)$
    & Continuum Lebesgue and Sobolev spaces; targets of the limit.
    & \S\ref{subsec:function_space_convergence} \\

% ── Measure theory ─────────────────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Measure theory}} \\
$\mu_\#$
    & Counting measure on $\Lambda_d$: $\mu_\#(A) = |A|$ for finite $A \subset \Lambda_d$.
    & \S\ref{subsec:counting_measure} \\
$\mu_{\#, \epsilon}$
    & $\epsilon$-rescaled counting measure on $\epsilon \Lambda_d$: $\mu_{\#, \epsilon} = \epsilon^d \mu_\#$.
    & \S\ref{subsec:counting_measure} \\

% ── Dimensional selection ─────────────────────────────────────────
\multicolumn{3}{@{}l}{\textit{Dimensional selection (\S\ref{sec:dimensional_selection})}} \\
$\delta p_\min$
    & Minimum probability quantum in the A=1 framework.  Hypothesised in this paper; empirically investigated in the companion repo \texttt{dcl-delta-p-min}.
    & \S\ref{subsec:dp_min_setup} \\
$d_\max$
    & Maximum admissible dimension under the hypothesis $\delta p_\min > 0$.  Theorem~\ref{thm:dp_min_dmax}: $d_\max = 1/\delta p_\min - 1$.
    & \S\ref{subsec:dp_min_dmax_corollary} \\

% ── Kinetic theory (motivating examples) ───────────────────────────
\multicolumn{3}{@{}l}{\textit{Kinetic theory (motivating examples, \S\ref{sec:motivating_examples})}} \\
$V$
    & Discrete velocity space attached to each lattice site.
    & \S\ref{subsec:boltzmann} \\
$f(x, v, t)$
    & Lattice kinetic distribution function.
    & \S\ref{subsec:boltzmann} \\
$Q[f]$
    & Boltzmann-shaped collision operator on the lattice.
    & \S\ref{subsec:boltzmann} \\
$H[f]$
    & Lattice $H$-functional: $H[f] = \sum_{x, v} f(x, v) \log f(x, v)$.
    & \S\ref{subsec:boltzmann} \\
$\rho$, $u$, $e$
    & Fluid density, velocity, energy.
    & \S\ref{subsec:fluid_limit} \\
