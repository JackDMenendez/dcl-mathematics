\begin{definition}
\label{def:diamond_progression}
The \emph{diamond progression in dimension $d$}, denoted
$\Lambda_d$, is the graph with

\begin{itemize}
\item vertex set $V_d = \mathbb{Z}^d$,
\item bipartition $V_d = V_d^+ \sqcup V_d^-$ where $V_d^+$ /
   $V_d^-$ are the sites whose coordinate sum is even / odd,
\item edges generated by the $d$ of $(d+1)$ regular-simplex vertex
   offsets (provisional; the explicit offset set is fixed in the
   construction below).
\end{itemize}
\end{definition}

\begin{proposition}
\label{prop:diamond_progression_bipartite}
For every $d \geq 2$, $\Lambda_d$ is a strict bipartite graph:
every edge in the edge set of
Definition~\ref{def:diamond_progression} connects a site in
$V_d^+$ to a site in $V_d^-$.
\end{proposition}

\subsection{Distinguishing Contrast with Chemical Diamond}
\label{subsec:contrast_chemical_diamond}

\placeholder{Why ``diamond progression'' as the name, when the
construction is not chemical diamond.  Chemical diamond has
coord~$4$ at every $d$ (face-centred-cubic structure repeated); the
diamond progression has $\text{coord}(d) = 2d$ (linear scaling).
The names share the connotation of ``tetrahedral / simplex
vertices'' but the constructions differ in their $d$-dependence
characteristically -- the chemical-diamond construction is a
\emph{repeated 3D motif}, while the diamond progression is a
\emph{single generative rule across all $d$}.}

The contrast becomes operational in
Section~\ref{sec:three_theorems}: chemical-diamond constructions
would not satisfy $\text{coord}(d) = 2d$, and consequently would
not exhibit the dimensional-selection corollary of
Section~\ref{sec:dimensional_selection}.

\subsection{Notation Conventions for the Rest of the Paper}
\label{subsec:lattice_notation}

Standing notation used throughout:

\begin{itemize}
\item $\Lambda_d$ -- the diamond progression in dimension $d$
      (Definition~\ref{def:diamond_progression}).
\item $V_d^+, V_d^-$ -- the bipartite colour classes (even / odd
      coordinate-sum parity).
\item $E_d$ -- the edge set of $\Lambda_d$.
\item $\epsilon$ -- the lattice spacing used in the
      discrete-to-continuum limit
      (Section~\ref{sec:continuum_limits}); $\epsilon \Lambda_d
      \subset \mathbb{R}^d$.
\item $\mathrm{d}$ -- the discrete exterior derivative
      (Section~\ref{sec:discrete_diff_geometry}); uses upright
      $\mathrm{d}$ to avoid notational collision with the
      dimension $d$.
\end{itemize}
